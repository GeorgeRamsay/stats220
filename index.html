My meme

I created this meme in particular because of the struggle I have using R. 

library(magick)

# Using image URL's to image
sad_cat <- image_read("https://scontent-akl1-1.xx.fbcdn.net/v/t39.30808-6/268087548_214085974226825_755029480199653790_n.jpg?_nc_cat=100&ccb=1-7&_nc_sid=09cbfe&_nc_ohc=wzYZid2I2RoAX9V8uhh&_nc_ht=scontent-akl1-1.xx&oh=00_AfAKleGplwXxR7q3_B9LexgxLc7Z_kT1fHZ_rfT4X5jsdA&oe=64130566") %>%
  image_scale(250)

happy_cat <- image_read("https://i.imgflip.com/cnudu.jpg") %>%
  image_scale(250)


# Creating space with set height and colour for 
sad_text <- image_blank(width = 250, 
                          height = 250, 
                          color = "#000000") %>%
  # Annotating the text onto the blank space
  image_annotate(text = "Using R",
                 color = "#FFFFFF",
                 size = 40,
                 font = "Impact",
                 gravity = "center")

# Repeated process as above adjusted for different text
happy_text <- image_blank(width = 250, 
                         height = 250, 
                         color = "#000000") %>%
  image_annotate(text = "Using python",
                 color = "#FFFFFF",
                 size = 40,
                 font = "Impact",
                 gravity = "center")

# Create rows to arrange images and text in a tidy mannor
first_row <- c(sad_cat, sad_text) %>%
  image_append()

second_row <- c(happy_cat, happy_text) %>%
  image_append()

# Join these rows together to complete image
boxes <- c(first_row, second_row) 
meme <-image_append(boxes, stack = TRUE)
meme
# Can export to a png
image_write(meme,"my_meme.png")

My animated GIF

I made this GIF because I like shurikens and wanted to animate it to spin. 

library(magick)

# Naming an image from its URL (setting size and background colour)
shuriken_image <- image_read("https://png.pngtree.com/png-clipart/20220725/ourmid/pngtree-shuriken-ninja-kunai-star-png-image_6060895.png") %>%
  image_scale(100) %>%
  image_background("white")

# creating each frame, where the shuriken rotates 45 degrees each frame
frame1 <- shuriken_image %>%
  image_extent("500x500")

frame2 <- shuriken_image %>%
  image_extent("500x500")%>%
  image_rotate(45)


frame3 <- shuriken_image %>%
  image_extent("500x500")%>%
  image_rotate(90)

frame4 <- shuriken_image %>%
  image_extent("500x500")%>%
  image_rotate(135)

frame5 <- shuriken_image %>%
  image_extent("500x500")%>%
  image_rotate(180)

frame6 <- shuriken_image %>%
  image_extent("500x500")%>%
  image_rotate(225)

frame7 <- shuriken_image %>%
  image_extent("500x500")%>%
  image_rotate(270)

frame8 <- shuriken_image %>%
  image_extent("500x500")%>%
  image_rotate(315)

# putting the frames in order using a vector
frames <- c(frame1, frame2, frame3, frame4, frame5, frame6, frame7, frame8)


# creating an animation making shuriken appear like it is spinning
animation <- image_animate(frames, fps = 5)
animation

# Can export to a gif
animation %>% 
  image_write("my_animation.gif")
